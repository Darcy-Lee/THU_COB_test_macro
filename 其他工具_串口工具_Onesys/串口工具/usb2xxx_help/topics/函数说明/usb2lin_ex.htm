概述 

新版本的USB2LIN接口函数说明，使用这些接口函数需要适配器固件大于或者等v1.5.30，若当前适配器固件版本低于这个版本，请参考《固件升级教程》升级固件。

下面是本文函数可能会用到的一些宏定义：

  1. //定义函数返回错误代码  
  2. #define LIN_EX_SUCCESS                           (0)   //函数执行成功  
  3. #define LIN_EX_ERR_NOT_SUPPORT       (-1)  //适配器不支持该函数  
  4. #define LIN_EX_ERR_USB_WRITE_FAIL  (-2)  //USB写数据失败  
  5. #define LIN_EX_ERR_USB_READ_FAIL    (-3)  //USB读数据失败  
  6. #define LIN_EX_ERR_CMD_FAIL               (-4)  //命令执行失败  
  7. #define LIN_EX_ERR_CH_NO_INIT          (-5)  //该通道未初始化  
  8. #define LIN_EX_ERR_READ_DATA            (-6)  //LIN读数据失败  
  9. #define LIN_EX_ERR_PARAMETER            (-7)  //函数参数传入有误 
  10. //校验类型定义  
  11. #define LIN_EX_CHECK_STD         0  //标准校验，不含PID  
  12. #define LIN_EX_CHECK_EXT         1  //增强校验，包含PID  
  13. #define LIN_EX_CHECK_USER      2  //自定义校验类型，需要用户自己计算并传入Check，不进行自动校验  
  14. #define LIN_EX_CHECK_NONE     3  //接收数据校验错误  
  15. #define LIN_EX_CHECK_ERROR   4  //接收数据校验错误  
  16. //校主从机定义   
  17. #define LIN_EX_MASTER              1//主机  
  18. #define LIN_EX_SLAVE                 0//从机  
  19. //帧类型定义
  20. #define LIN_EX_MSG_TYPE_UN       0   //未知类型  
  21. #define LIN_EX_MSG_TYPE_MW      1   //主机向从机发送数据  
  22. #define LIN_EX_MSG_TYPE_MR       2   //主机从从机读取数据  
  23. #define LIN_EX_MSG_TYPE_SW      3   //从机发送数据  
  24. #define LIN_EX_MSG_TYPE_SR       4   //从机接收数据  
  25. #define LIN_EX_MSG_TYPE_BK       5   //只发送BREAK信号，若是反馈回来的数据，表明只检测到BREAK信号  
  26. #define LIN_EX_MSG_TYPE_SY       6   //表明检测到了BREAK，SYNC信号  
  27. #define LIN_EX_MSG_TYPE_ID       7   //表明检测到了BREAK，SYNC，PID信号  
  28. #define LIN_EX_MSG_TYPE_DT       8   //表明检测到了BREAK，SYNC，PID,DATA信号  
  29. #define LIN_EX_MSG_TYPE_CK       9   //表明检测到了BREAK，SYNC，PID,DATA,CHECK信号  
  30.   
  31. typedef struct _LIN_EX_MSG{  
  32.     unsigned int  Timestamp;    //接收数据时代表时间戳，单位为100us，发送数据时代表该帧发送完毕后的延时时间，单位为毫秒 
  33.     unsigned char MsgType;      //帧类型  
  34.     unsigned char CheckType;   //校验类型  
  35.     unsigned char DataLen;      //LIN数据段有效数据字节数  
  36.     unsigned char Sync;            //固定值，0x55  
  37.     unsigned char PID;             //帧ID         
  38.     unsigned char Data[8];      //数据  
  39.     unsigned char Check;        //校验,只有校验数据类型为LIN_EX_CHECK_USER的时候才需要用户传入数据  
  40.     unsigned char Reserve0;  
  41.     unsigned char Reserve1;  
  42. }LIN_EX_MSG,*PLIN_EX_MSG;  

LIN_EX_Init

描述：

初始化配置USB2LIN适配器。

原型：

int WINAPI  LIN_EX_Init(int DevHandle,unsigned char LINIndex,unsigned int BaudRate,unsigned char MasterMode); 

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

BaudRate  LIN通信波特率，填9600,19200,10000,20000等任意值，最大可支持到100000.

MasterMode  主从机配置，0-初始化为从机，1-初始化为主机（会自动使能LIN总线1K上拉电阻）。

返回值：

函数执行状态，0表示函数执行成功，其他值表示函数执行失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2lin_ex.h"  
  6.   
  7. int DevHandle[10];  
  8. int LINMasterIndex = 0;  
  9. int DevIndex = 0;  
  10. bool state;  
  11. int ret;  
  12.   
  13. //扫描查找设备  
  14. ret = USB_ScanDevice(DevHandle);  
  15. if(ret <= 0){  
  16.     printf("No device connected!\n");  
  17.     return 0;  
  18. }  
  19. //打开设备  
  20. state = USB_OpenDevice(DevHandle[DevIndex]);  
  21. if(!state){  
  22.     printf("Open device error!\n");  
  23.     return 0;  
  24. }  
  25. //初始化配置LIN  
  26. ret = LIN_EX_Init(DevHandle[DevIndex],LINMasterIndex,9600,1);  
  27. if(ret != LIN_EX_SUCCESS){  
  28.     printf("Config LIN failed!\n");  
  29.     return 0;  
  30. }else{  
  31.     printf("Config LIN Success!\n");  
  32. }  

LIN_EX_MasterSync

描述：

主机模式发送或读取数据。主机通过消息帧的方式收发数据，消息帧类型可以设置为发送BREAK信号（LIN_EX_MSG_TYPE_BK），主机写数据（LIN_EX_MSG_TYPE_MW），主机读数据（LIN_EX_MSG_TYPE_MR）。

再执行这些消息的时候，适配器会自动监控LIN总线上的数据，然后通过该函数的pOutMsg参数将收到的数据返回上层，上层函数可以对比接收到的数据和发送出去的数据就能判断数据是否成功发送。

原型：

  1. int WINAPI  LIN_EX_MasterSync(int DevHandle,unsigned char LINIndex,LIN_EX_MSG *pInMsg,LIN_EX_MSG *pOutMsg,unsigned int MsgLen);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

pInMsg  输入的消息帧指针。

pOutMsg  输出的消息帧指针。

MsgLen  输入的消息帧数。

返回值：

小于0表示函数调用出错，大于0表示接收到LIN总线上数据帧数。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2lin_ex.h"  
  6.   
  7. int DevHandle[10];  
  8. int LINMasterIndex = 0;  
  9. int DevIndex = 0;  
  10. bool state;  
  11. int ret;  
  12.   
  13. char *MSGTypeStr[]={"UN","MW","MR","SW","SR","BK","SY","ID","DT","CK"};  
  14. char *CKTypeStr[]={"STD","EXT","USER","NONE","ERROR"};  
  15.   
  16. LIN_EX_MSG LINMsg[5];  
  17. LIN_EX_MSG LINOutMsg[10];  
  18. unsigned int MsgLen = 5;  
  19. LINMsg[0].MsgType = LIN_EX_MSG_TYPE_BK;  
  20. LINMsg[0].Timestamp = 10;//发送该帧数据之后的延时时间，最小建议设置为1  
  21. for(int f=1;f<MsgLen;f++){  
  22.     LINMsg[f].MsgType = LIN_EX_MSG_TYPE_MW;  
  23.     LINMsg[f].DataLen = 8;  
  24.     for(int i=0;i<LINMsg[1].DataLen;i++){  
  25.         LINMsg[f].Data[i] = (f<<4)|i;  
  26.     }  
  27.     LINMsg[f].Timestamp = 10;//发送该帧数据之后的延时时间，最小建议设置为1  
  28.     LINMsg[f].CheckType = LIN_EX_CHECK_EXT;  
  29.     LINMsg[f].PID = f+1;  
  30. }  
  31.   
  32. ret = LIN_EX_MasterSync(DevHandle[DevIndex],LINMasterIndex,LINMsg,LINOutMsg,MsgLen,0);  
  33. if(ret < LIN_EX_SUCCESS){  
  34.     printf("MasterSync LIN failed!\n");  
  35.     return 0;  
  36. }else{  
  37.     printf("MsgLen = %d\n",ret);  
  38.     for(int i=0;i<ret;i++){  
  39.         printf("%s SYNC[%02X] PID[%02X] ",MSGTypeStr[LINOutMsg[i].MsgType],LINOutMsg[i].Sync,LINOutMsg[i].PID);  
  40.         for(int j=0;j<LINOutMsg[i].DataLen;j++){  
  41.             printf("%02X ",LINOutMsg[i].Data[j]);  
  42.         }  
  43.         printf("[%s][%02X] [%02d:%02d:%02d.%03d]\n",CKTypeStr[LINOutMsg[i].CheckType],LINOutMsg[i].Check,(LINOutMsg[i].Timestamp/36000000)%60,(LINOutMsg[i].Timestamp/600000)%60,(LINOutMsg[i].Timestamp/10000)%60,(LINOutMsg[i].Timestamp/10)%1000);  
  44.     }  
  45. }

LIN_EX_MasterWrite

描述：

主机模式发送数据给从机，调用改函数后，适配器会发送同步间隔+0x55+PID+数据+校验到总线上。

原型：

int WINAPI  LIN_EX_MasterWrite(int DevHandle,unsigned char LINIndex,unsigned char PID,unsigned char *pData,unsigned char DataLen,unsigned char CheckType);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

PID  发送LIN数的ID，只传入低6位ID值即可，底层会自动计算并填充高2位的校验数据，取值范围为0x00~0x3F。

pData  需要发送数据的指针，最多只能发送8字节数据。

DataLen  发送数据字节数，最大只能取值8，若设置为0，调用改函数后只发送帧头，校验也不会被发送。

CheckType  校验类型，0-标准校验，校验不包含PID，1-增强校验，校验包含PID。

返回值：

小于0表示函数调用出错，等于0表示函数调用成功。

原型：

int WINAPI  LIN_EX_MasterRead(int DevHandle,unsigned char LINIndex,unsigned char PID,unsigned char *pData);

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

PID  发送LIN数的ID，只传入低6位ID值即可，底层会自动计算并填充高2位的校验数据，取值范围为0x00~0x3F。

pData  存储从机返回数据的数据缓冲区指针，建议将缓冲区空件分配8字节空间。

返回值：

小于0表示函数调用出错，大于或者等于0，表示读到的数据字节数。

LIN_EX_SlaveSetIDMode

描述：

从机模式下设置每个ID的操作模式，比如设置为从机接收数据或者从机发送数据。需要监控总线上的数据，可以将所有ID都设置为从机读数据（初始化为从机后默认为该模式）。

原型：

  1. int WINAPI  LIN_EX_SlaveSetIDMode(int DevHandle,unsigned char LINIndex,LIN_EX_MSG *pLINMsg,unsigned int MsgLen);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

pLINMsg  输入的消息帧指针。

MsgLen  输入的消息帧数。

返回值：

等于0表示函数调用成功，其他值表示函数调用失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2lin_ex.h"  
  6.   
  7. int DevHandle[10];  
  8. int LINSlaveIndex= 0;  
  9. int DevIndex = 0;  
  10. bool state;  
  11. int ret;  
  12.   
  13. //初始化配置LIN  
  14. ret = LIN_EX_Init(DevHandle[DevIndex],LINSlaveIndex,9600,0);  
  15. if(ret != LIN_EX_SUCCESS){  
  16.     printf("Config LIN failed!\n");  
  17.     return 0;  
  18. }else{  
  19.     printf("Config LIN Success!\n");  
  20. }  
  21. printf("Start Get LIN Data...\n");  
  22. //设置ID为LIN_EX_MSG_TYPE_SW模式，这样主机就可以读取到数据  
  23. LIN_EX_MSG LINSlaveMsg[10];  
  24. for(int i=0;i<10;i++){  
  25.     LINSlaveMsg[i].PID = i;  
  26.     LINSlaveMsg[i].CheckType = LIN_EX_CHECK_EXT;  
  27.     LINSlaveMsg[i].DataLen = 7;  
  28.     for(int j=0;j<LINSlaveMsg[i].DataLen;j++){  
  29.         LINSlaveMsg[i].Data[j]=(i<<4)|j;  
  30.     }  
  31.     LINSlaveMsg[i].MsgType = LIN_EX_MSG_TYPE_SW;//从机发送数据模式  
  32. }  
  33. ret = LIN_EX_SlaveSetIDMode(DevHandle[DevIndex],LINSlaveIndex,LINSlaveMsg,10);  
  34. if(ret != LIN_EX_SUCCESS){  
  35.     printf("Config LIN ID Mode failed!\n");  
  36.     return 0;  
  37. }else{  
  38.     printf("Config LIN ID Mode Success!\n");  
  39. }

LIN_EX_SlaveGetData

描述：

从机模式下获取从总线上接收到数据缓冲区中的数据，适配器内部有2个数据接收缓冲区，读取数据之后会自动切换另外一个缓冲区，保证数据不会被冲突。

原型：

  1. int WINAPI  LIN_EX_SlaveGetData(int DevHandle,unsigned char LINIndex,LIN_EX_MSG *pLINMsg);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

pLINMsg  存储接收到的数据缓冲区指针，建议将缓冲区设置大一点，以免读取到的数据太多导致缓冲区溢出。

返回值：

小于0表示函数调用出错，大于0表示接收到LIN总线上数据帧数。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2lin_ex.h"  
  6.   
  7. int DevHandle[10];  
  8. int LINSlaveIndex= 0;  
  9. int DevIndex = 0;  
  10. bool state;  
  11. int ret;  
  12.   
  13. char *MSGTypeStr[]={"UN","MW","MR","SW","SR","BK","SY","ID","DT","CK"};  
  14. char *CKTypeStr[]={"STD","EXT","USER","NONE","ERROR"};  
  15.   
  16. //循环获取接收到的数据，该操作可以用作LIN总线数据监控  
  17. while(1)  
  18. {  
  19.     LIN_EX_MSG LINMsg[1024];//缓冲区尽量大一点，防止益处  
  20.     int ret = LIN_EX_SlaveGetData(DevHandle[DevIndex],LINSlaveIndex,LINMsg);  
  21.     for(int i=0;i<ret;i++){  
  22.         printf("%s SYNC[%02X] PID[%02X] ",MSGTypeStr[LINMsg[i].MsgType],LINMsg[i].Sync,LINMsg[i].PID);  
  23.         for(int j=0;j<LINMsg[i].DataLen;j++){  
  24.             printf("%02X ",LINMsg[i].Data[j]);  
  25.         }  
  26.         printf("[%s][%02X] [%02d:%02d:%02d.%03d]\n",CKTypeStr[LINMsg[i].CheckType],LINMsg[i].Check,(LINMsg[i].Timestamp/36000000)%60,(LINMsg[i].Timestamp/600000)%60,(LINMsg[i].Timestamp/10000)%60,(LINMsg[i].Timestamp/10)%1000);  
  27.     }  
  28.     Sleep(1000);  
  29. }  

LIN_EX_MasterStartSch

描述：

主机模式下将调度表放到适配器内部并启动执行调度表，所谓调度表即为主机模式下的数据读写操作列表。

原型：

  1. int WINAPI  LIN_EX_MasterStartSch(int DevHandle,unsigned char LINIndex,LIN_EX_MSG *pLINMsg,unsigned int MsgLen);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

pLINMsg  待发送的调度表数据缓冲区指针。

返回值：

等于0表示函数调用成功，其他值表示函数调用失败。

示例：

  1. #include <stdio.h>  
  2. #include "usb2lin_ex.h"  
  3.   
  4. LIN_EX_MSG LINMsg[6];  
  5. LIN_EX_MSG LINOutMsg[10];  
  6. unsigned int MsgLen = 6;  
  7. LINMsg[0].MsgType = LIN_EX_MSG_TYPE_BK;  
  8. LINMsg[0].Timestamp = 20;//当前帧发送时，下一帧数据发送的时间间隔，单位为毫秒  
  9. for(int f=1;f<MsgLen;f++){  
  10.     LINMsg[f].MsgType = LIN_EX_MSG_TYPE_MW;  
  11.     LINMsg[f].DataLen = 8;  
  12.     for(int i=0;i<LINMsg[1].DataLen;i++){  
  13.         LINMsg[f].Data[i] = (f<<4)|i;  
  14.     }  
  15.     LINMsg[f].Timestamp = 20+f*10;//当前帧发送时，下一帧数据发送的时间间隔，单位为毫秒  
  16.     LINMsg[f].CheckType = LIN_EX_CHECK_EXT;  
  17.     LINMsg[f].PID = f+1;  
  18. }  
  19. //ID=6，该帧为主机读数据模式，也就是主机发送帧头，从机返回数据  
  20. LINMsg[5].MsgType = LIN_EX_MSG_TYPE_MR;  
  21. LINMsg[5].PID = 6;  
  22. LINMsg[5].Timestamp = 20;  
  23. //启动调度表  
  24. ret = LIN_EX_MasterStartSch(DevHandle[DevIndex],LINMasterIndex,LINMsg,MsgLen);  
  25. if(ret < LIN_EX_SUCCESS){  
  26.     printf("Master schedule start failed!\n");  
  27.     return 0;  
  28. }else{  
  29.     printf("Master schedule start success!\n");  
  30. }  

LIN_EX_MasterStopSch

描述：

停止正在自动执行的主机调度表。

原型：

  1. int WINAPI  LIN_EX_MasterStopSch(int DevHandle,unsigned char LINIndex);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

LIN_EX_MasterGetSch

描述：

获取调度表数据，比如调度表中有主机读数据的操作，调用该函数可以获取到主机从从机读取到的数据。

原型：

  1. int WINAPI  LIN_EX_MasterGetSch(int DevHandle,unsigned char LINIndex,LIN_EX_MSG *pLINMsg);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

LINIndex  LIN通道索引号，填0或者1，若只有一个通道LIN，则填0.

pLINMsg  存储接收到的数据缓冲区指针，该缓冲区大小必须大于或者等于启动调度表时传递给适配器的调度表大小，否则可能会出现缓冲区溢出，导致程序异常关闭。

返回值：

小于0表示函数调用出错，大于0表示获取到的调度表大小。

LIN_EX_CtrlPowerOut

描述：

控制适配器的电源输出口输出12V的电压，若LIN总线需要其他电平电压，则可以关闭输出，直接将输出电压接口接到外接电源即可。

原型：

  1. int WINAPI  LIN_EX_CtrlPowerOut(int DevHandle,unsigned char State);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

State  电平输出状态，0-禁止输出，当前电压为USB接口电压（5V）；1-输出12V电压

返回值：

等于0表示函数调用成功，其他值表示函数调用失败。
