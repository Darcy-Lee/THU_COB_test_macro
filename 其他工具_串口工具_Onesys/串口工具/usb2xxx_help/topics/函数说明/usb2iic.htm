概述

USB2XXX有8路IIC/I2C信号输出，每路最大时钟频率可达400Kbps,USB2XXX的IIC/I2C总线上不带上拉电阻，在实际使用的时候需要外接上拉电阻，或者软件上使能内部上拉电阻，总线上拉电压范围为1.8V到5V。U2B2XXX的IIC/I2C支持主机模式和从机模式（只有通道0和通道1支持从机模式）。

下面是本文函数可能会用到的一些宏定义：

  1. //定义初始化IIC的数据类型  
  2. typedef struct _IIC_CONFIG{  
  3.   unsigned int    ClockSpeedHz;   //IIC时钟频率:单位为Hz  
  4.   unsigned short  OwnAddr;        //USB2XXX为从机时自己的地址  
  5.   unsigned char   Master;            //主从选择控制:0-从机，1-主机  
  6.   unsigned char   AddrBits;         //从机地址模式，7-7bit模式，10-10bit模式  
  7.   unsigned char   EnablePu;       //使能引脚芯片内部上拉电阻，若不使能，则I2C总线上必须接上拉电阻  
  8. }IIC_CONFIG,*PIIC_CONFIG;  
  9.   
  10. //定义函数返回错误代码  
  11. #define IIC_SUCCESS                          (0)   //函数执行成功  
  12. #define IIC_ERR_NOT_SUPPORT       (-1)  //适配器不支持该函数  
  13. #define IIC_ERR_USB_WRITE_FAIL  (-2)  //USB写数据失败  
  14. #define IIC_ERR_USB_READ_FAIL    (-3)  //USB读数据失败  
  15. #define IIC_ERR_CMD_FAIL               (-4)  //命令执行失败  
  16. #define IIC_ERR_PARA_ERROR          (-5)  //参数传入错误  
  17. //定义IIC函数返回错误代码  
  18. #define IIC_ERROR_SUCCESS           0   //操作成功  
  19. #define IIC_ERROR_CHANNEL           1   //该通道不支持该函数  
  20. #define IIC_ERROR_BUSY                 2   //总线忙  
  21. #define IIC_ERROR_START_FAILD   3   //启动总线失败  
  22. #define IIC_ERROR_TIMEOUT          4   //超时  
  23. #define IIC_ERROR_NACK                5   //从机无应答  
  24. #define IIC_ERROR_PEC                   6   //PEC校验失败  

IIC_Init

描述：

初始化配置IIC/I2C总线。

原型：

  1. int WINAPI IIC_Init(int DevHandle,int IICIndex, PIIC_CONFIG pConfig);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

pConfig   IIC/I2C初始化配置参数结构指针，具体取值说明可参考概述说明。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7.    
  8. int IICIndex = 0;  
  9. int  ret;  
  10. //初始化配置I2C  
  11. IIC_CONFIG IIC_Config;  
  12. IIC_Config.AddrBits = 7;            //7bit地址模式  
  13. IIC_Config.ClockSpeedHz = 400000;     //时钟频率400KHz  
  14. IIC_Config.EnablePu = 1;                    //使能内部上拉电阻  
  15. #if SLAVE_WRITE_TEST||SLAVE_READ_TEST  
  16. IIC_Config.Master = 0;              //从机模式  
  17. IIC_Config.OwnAddr = 0x71;          //从机地址  
  18. #else  
  19. IIC_Config.Master = 1;              //主机模式  
  20. #endif  
  21. ret = IIC_Init(DevHandle,IICIndex,&IIC_Config);  
  22. if(ret != IIC_SUCCESS){  
  23.     printf("Config IIC failed!\n");  
  24.     return 0;  
  25. }else{  
  26.     printf("Config IIC IIC_SUCCESS!\n");  
  27. }  

IIC_GetSlaveAddr

描述：

获取IIC/I2C总线上已经连接好的从机设备地址列表，使用此函数可以检测当前总线上连接的器件是否能正常工作。

原型：

  1. int WINAPI IIC_GetSlaveAddr(int DevHandle,int IICIndex,short *pSlaveAddr);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

pSlaveAddr  IIC/I2C总线上能正常响应的从设备地址列表指针。

返回值：

大于0为扫描到的从机数量，小于0则函数调用出错。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.   
  7. int IICIndex = 0;  
  8. int  ret;  
  9. //获取总线上能应答的从机地址  
  10. short SlaveAddrs[128];  //地址存储缓冲区  
  11. int SlaveAddrNum = 0;   //返回应答的地址个数  
  12. SlaveAddrNum = IIC_GetSlaveAddr(DevHandle,IICIndex,SlaveAddrs);  
  13. if(SlaveAddrNum >= 0){  
  14.     printf("Get %d slave address!\n",SlaveAddrNum);  
  15.     for(int i=0;i<SlaveAddrNum;i++){  
  16.         printf("%02X ",SlaveAddrs[i]);  
  17.     }  
  18.     printf("\n");  
  19. }else{  
  20.     printf("Get slave address error!\n");  
  21.     return 0;  
  22. }  

IIC_WriteBytes

描述：

主机模式向IIC/I2C总线上写数据，调用该函数后，USB2XXX会先发送从机地址，若从机正常应答之后再发送指定字节数数据。

对应IIC/I2C本身来说是没有定义子地址或者内部寄存器地址概念的，但是大多数芯片在写数据时都要求先写一个内部寄存器地址，再发送要写入该寄存器的数据，那么在使用此函数时，只需要把内部寄存器地址放到发送的数据前面即可。

原型：

  1. int WINAPI IIC_WriteBytes(int DevHandle,int IICIndex,short SlaveAddr,unsigned char *pWriteData,int WriteLen,int TimeOutMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

SlaveAddr  IIC/I2C从设备地址，不包含读写位，比如包含了读写位的从机地址是0xA0，那么不包含读写位的从机地址就是0xA0>>1，也就是0x50.

pWriteData  待发送的数据缓冲区指针。

WriteLen  待发送数据的字节数。

TimeOutMs  IIC/I2C总线若不能正常响应时，其等待时间，单位为毫秒。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7. int IICIndex = 0;  
  8. int  ret;  
  9. //IIC 写数据  
  10. unsigned char WriteBuffer[8]={0x10,0x02,0x03,0x04,0x05,0x06,0x07,0x08};  
  11. ret = IIC_WriteBytes(DevHandle,IICIndex,0x50,WriteBuffer,sizeof(WriteBuffer),10);  
  12. if(ret != IIC_SUCCESS){  
  13.     printf("Write IIC failed!\n");  
  14.     printf("Error Code:%d\n",ret);  
  15.     return 0;  
  16. }else{  
  17.     printf("Write IIC IIC_SUCCESS!\n");  
  18. }

原型：

  1. int WINAPI IIC_ReadBytes(int DevHandle,int IICIndex,short SlaveAddr,unsigned char *pReadData,int ReadLen,int TimeOutMs);

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

SlaveAddr  IIC/I2C从设备地址，不包含读写位，比如包含了读写位的从机地址是0xA0，那么不包含读写位的从机地址就是0xA0>>1，也就是0x50.

pReadData  存储从从机返回的数据缓冲区首地址。

ReadLen  待读取数据的字节数。

TimeOutMs  IIC/I2C总线若不能正常响应时，其等待时间，单位为毫秒。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.   
  7. int IICIndex = 0;  
  8. int  ret;  
  9. //IIC 读数据  
  10. unsigned char ReadBuffer[8];  
  11. ret = IIC_ReadBytes(DevHandle,IICIndex,0x50,ReadBuffer,sizeof(ReadBuffer),10);  
  12. if(ret != IIC_SUCCESS){  
  13.     printf("Read IIC failed!\n");  
  14.     printf("Error Code:%d\n",ret);  
  15.     return 0;  
  16. }else{  
  17.     printf("Read IIC IIC_SUCCESS!\n");  
  18.     printf("Read Data:\n");  
  19.     for(int i=0;i<sizeof(ReadBuffer);i++){  
  20.         printf("%02X ",ReadBuffer[i]);  
  21.     }  
  22.     printf("\n");  
  23. }  

IIC_WriteReadBytes

描述：

主机模式向IIC/I2C总线发送数据之后再读取指定字节数数据，调用该函数后，USB2XXX会先发送从机地址，然后发送指定的数据，然后再发送从机地址，再从从机读取数据，该函数在操作带内部寄存器地址的设备的时候非常有用。比如IIC/I2C接口的EEPROM。

原型：

  1. int WINAPI IIC_WriteReadBytes(int DevHandle,int IICIndex,short SlaveAddr,unsigned char *pWriteData,int WriteLen,unsigned char *pReadData,int ReadLen,int TimeOutMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

SlaveAddr  IIC/I2C从设备地址，不包含读写位，比如包含了读写位的从机地址是0xA0，那么不包含读写位的从机地址就是0xA0>>1，也就是0x50.

pWriteData  待发送的数据缓冲区首地址。

WriteLen  待发送数据的字节数。

pReadData  存储从从机返回的数据缓冲区首地址。

ReadLen  待读取数据的字节数。

TimeOutMs  IIC/I2C总线若不能正常响应时，其等待时间，单位为毫秒。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7. int IICIndex = 0;  
  8. int ret;  
  9. unsigned char WriteBuffer[8];  
  10. unsigned char ReadBuffer[8];  
  11. //IIC 写读数据。也就是先发送数据，然后再次产生START信号，再读数据  
  12. //注意：对于每个I2C设备，必须按照设备要求正确读写，否则可能会导致设备工作不正常，从而导致无法读写数据  
  13. WriteBuffer[0] = 0x08;  
  14. ret = IIC_WriteReadBytes(DevHandle,IICIndex,0x50,WriteBuffer,1,ReadBuffer,sizeof(ReadBuffer),1000);  
  15. if(ret != IIC_SUCCESS){  
  16.     printf("WriteRead IIC failed!\n");  
  17.     printf("Error Code:%d\n",ret);  
  18.     return 0;  
  19. }else{  
  20.     printf("WriteRead IIC IIC_SUCCESS!\n");  
  21.     printf("Read Data:\n");  
  22.     for(int i=0;i<sizeof(ReadBuffer);i++){  
  23.         printf("%02X ",ReadBuffer[i]);  
  24.     }  
  25.     printf("\n");  
  26. }  

IIC_SlaveWriteBytes 

描述：

从机模式写数据，调用该函数后，数据会存放到USB2XXX内部数据缓冲区，然后等待主机产生读数据时序，地址匹配后再将数据移位发送出去。

原型：

  1. int WINAPI IIC_SlaveWriteBytes(int DevHandle,int IICIndex,unsigned char *pWriteData,int WriteLen,int TimeOutMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

pWriteData  待发送的数据缓冲区首地址。

WriteLen  待发送数据的字节数。

TimeOutMs  等待数据被读取的超时时间，若为0则不用等待数据发送完毕立即返回，大于0则等待数据发送完毕或者超时之后再返回。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7. int IICIndex = 0;  
  8. int ret;  
  9. //IIC 从机模式写数据  
  10. unsigned char SlaveWriteBuffer[256];  
  11. for(int i=0;i<sizeof(SlaveWriteBuffer);i++){  
  12.     SlaveWriteBuffer[i] = i;  
  13. }  
  14. for(int i=0;i<(sizeof(SlaveWriteBuffer)/8);i++){  
  15.     ret = IIC_SlaveWriteBytes(DevHandle,IICIndex,&SlaveWriteBuffer[i*8],8,10000);  
  16.     if(ret < 0){  
  17.         printf("Slave write data error!\n");  
  18.         break;  
  19.     }else if(ret == 0){  
  20.         printf("Slave write data IIC_SUCCESS!\n");  
  21.     }else if(ret > 0){  
  22.         printf("Slave write data IIC_SUCCESS! have %d byte data remain\n",ret);  
  23.     }  
  24. }  

IIC_SlaveReadBytes

描述：

获取在从机模式下接收到的数据。

原型：

  1. int WINAPI IIC_SlaveReadBytes(int DevHandle,int IICIndex,unsigned char *pReadData,int TimeOutMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

pReadData  存储IIC/I2C在从机模式下接收到的数据缓冲区首地址。

TimeOutMs  等待接收数据超时参数，若为0则立即返回，大于0则接收到数据或者超时后返回。

返回值：

小于0则该函数调用出错，否则返回读取到的数据字节数。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7. int IICIndex = 0;  
  8. int ret;  
  9. //IIC 从机模式读数据  
  10. unsigned char SlaveReadBuffer[256];  
  11. for(int i=0;i<16;i++){  
  12.     ret = IIC_SlaveReadBytes(DevHandle,IICIndex,SlaveReadBuffer,10000);  
  13.     if(ret < 0){  
  14.         printf("Slave read data error!\n");  
  15.         break;  
  16.     }else if(ret == 0){  
  17.         printf("Slave read data IIC_SUCCESS! but no data\n");  
  18.     }else if(ret > 0){  
  19.         printf("Slave read data IIC_SUCCESS! have %d byte data have read\n",ret);  
  20.         for(int i=0;i<ret;i++){  
  21.             printf("%02X ",SlaveReadBuffer[i]);  
  22.         }  
  23.         printf("\n");  
  24.     }  
  25. }  
  26. printf("Slave read end!\n");

IIC_WriteBytesOfEvent

描述：

等到事件后主机模式向IIC/I2C总线上写数据，调用该函数后，USB2XXX会先发送从机地址，若从机正常应答之后再发送指定字节数数据。

原型：

  1. int WINAPI IIC_WriteBytesOfEvent(int DevHandle,int IICIndex,short SlaveAddr,unsigned char *pWriteData,int WriteLen,int EventPin,unsigned char EventType,int TimeOutOfEventMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

SlaveAddr  IIC/I2C从设备地址，不包含读写位，比如包含了读写位的从机地址是0xA0，那么不包含读写位的从机地址就是0xA0>>1，也就是0x50.

pWriteData  待发送的数据缓冲区首地址。

WriteLen  待发送数据的字节数。

EventPin  等待事件输入的GPIO引脚，每个bit位对应一个引脚，比如P0为0x01，P2为0x04，P15为0x8000。

EventType   等待件类型，0x00-低电平，0x11-高电平，0x10-下降沿，0x01上升沿。

TimeOutOfEventMs  等待事件超时时间，单位为毫秒，等待超时后该函数直接返回，不执行任何操作。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

IIC_ReadBytesOfEvent

描述：

等到事件后主机模式向IIC/I2C从机读取数据，调用该函数后，USB2XXX会先发送从机地址，若从机正常响应了则会进入主机接收模式从从机读取指定字节数数据。

原型：

  1. int WINAPI IIC_ReadBytesOfEvent(int DevHandle,int IICIndex,short SlaveAddr,unsigned char *pReadData,int ReadLen,int EventPin,unsigned char EventType,int TimeOutOfEventMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

SlaveAddr  IIC/I2C从设备地址，不包含读写位，比如包含了读写位的从机地址是0xA0，那么不包含读写位的从机地址就是0xA0>>1，也就是0x50.

pReadData  存储从从机返回的数据缓冲区首地址。

ReadLen  待读取数据的字节数。

EventPin  等待事件输入的GPIO引脚，每个bit位对应一个引脚，比如P0为0x01，P2为0x04，P15为0x8000。

EventType   等待件类型，0x00-低电平，0x11-高电平，0x10-下降沿，0x01上升沿。

TimeOutOfEventMs  等待事件超时时间，单位为毫秒，等待超时后该函数直接返回，不执行任何操作。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7. int IICIndex = 0;  
  8. int  ret;  
  9. //IIC 读数据  
  10. unsigned char ReadBuffer[8];  
  11. ret = IIC_ReadBytesOfEvent(DevHandle,IICIndex,0x50,ReadBuffer,sizeof(ReadBuffer),0x01,0x10,100);  
  12. if(ret != IIC_SUCCESS){  
  13.     printf("Read IIC failed!\n");  
  14.     printf("Error Code:%d\n",ret);  
  15.     return 0;  
  16. }else{  
  17.     printf("Read IIC IIC_SUCCESS!\n");  
  18.     printf("Read Data:\n");  
  19.     for(int i=0;i<sizeof(ReadBuffer);i++){  
  20.         printf("%02X ",ReadBuffer[i]);  
  21.     }  
  22.     printf("\n");  
  23. }  

IIC_WriteReadBytesOfEvent

描述：

等到事件后主机模式向IIC/I2C总线发送数据之后再读取指定字节数数据，调用该函数后，USB2XXX会先发送从机地址，然后发送指定的数据，然后再发送从机地址，再从从机读取数据，该函数在操作带内部寄存器地址的设备的时候非常有用。比如IIC/I2C接口的EEPROM。

原型：

  1. int WINAPI IIC_WriteReadBytesOfEvent(int DevHandle,int IICIndex,short SlaveAddr,unsigned char *pWriteData,int WriteLen,unsigned char *pReadData,int ReadLen,int EventPin,unsigned char EventType,int TimeOutOfEventMs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数获得。

IICIndex  IIC/I2C通道索引号，每个通道对应的管脚可参考《引脚定义说明文档 》。

SlaveAddr  IIC/I2C从设备地址，不包含读写位，比如包含了读写位的从机地址是0xA0，那么不包含读写位的从机地址就是0xA0>>1，也就是0x50.

pWriteData  待发送的数据缓冲区首地址。

WriteLen  待发送数据的字节数。

pReadData  存储从从机返回的数据缓冲区首地址。

ReadLen  待读取数据的字节数。

EventPin  等待事件输入的GPIO引脚，每个bit位对应一个引脚，比如P0为0x01，P2为0x04，P15为0x8000。

EventType   等待件类型，0x00-低电平，0x11-高电平，0x10-下降沿，0x01上升沿。

TimeOutOfEventMs  等待事件超时时间，单位为毫秒，等待超时后该函数直接返回，不执行任何操作。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2iic.h"  
  6.    
  7. int IICIndex = 0;  
  8. int ret;  
  9. unsigned char WriteBuffer[8];  
  10. unsigned char ReadBuffer[8];  
  11. //IIC 写读数据。也就是先发送数据，然后再次产生START信号，再读数据  
  12. //注意：对于每个I2C设备，必须按照设备要求正确读写，否则可能会导致设备工作不正常，从而导致无法读写数据  
  13. WriteBuffer[0] = 0x08;  
  14. ret = IIC_WriteReadBytesOfEvent(DevHandle,IICIndex,0x50,WriteBuffer,1,ReadBuffer,sizeof(ReadBuffer),0x01,0x10,100);  
  15. if(ret != IIC_SUCCESS){  
  16.     printf("WriteRead IIC failed!\n");

  18.     return 0;  
  19. }else{  
  20.     printf("WriteRead IIC IIC_SUCCESS!\n");  
  21.     printf("Read Data:\n");  
  22.     for(int i=0;i<sizeof(ReadBuffer);i++){  
  23.         printf("%02X ",ReadBuffer[i]);  
  24.     }  
  25.     printf("\n");
