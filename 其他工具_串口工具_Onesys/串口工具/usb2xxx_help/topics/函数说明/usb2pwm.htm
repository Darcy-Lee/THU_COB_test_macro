概述

USB2XXX有8路独立的PWM输出，每路的频率和占空比都可独立调节，最大输出频率可达100MHz,最小频率可达1Hz,占空比调节精度可自己设定，脉冲极性可调。

可以在输出PWM的同时调整PWM占空比和频率。

下面是本文函数可能会用到的一些宏定义：

  1. //定义初始化PWM的数据类型  
  2. typedef struct _PWM_CONFIG{  
  3.   unsigned short Prescaler[8];      //预分频器  
  4.   unsigned short Precision[8];      //占空比调节精度,实际频率 = 200MHz/(Prescaler*Precision)  
  5.   unsigned short Pulse[8];           //占空比，实际占空比=(Pulse/Precision)*100%  
  6.   unsigned short Phase[8];          //波形相位，取值0到Precision-1
  7.   unsigned char  Polarity[8];        //波形极性，取值0或者1  
  8.   unsigned char  ChannelMask;    //通道号，若要使能某个通道，则对应位为1，最低位对应通道0  
  9. }PWM_CONFIG,*PPWM_CONFIG;  
  10.   
  11. //定义函数返回错误代码  
  12. #define PWM_SUCCESS                           (0)   //函数执行成功  
  13. #define PWM_ERR_NOT_SUPPORT        (-1)  //适配器不支持该函数  
  14. #define PWM_ERR_USB_WRITE_FAIL   (-2)  //USB写数据失败  
  15. #define PWM_ERR_USB_READ_FAIL     (-3)  //USB读数据失败  
  16. #define PWM_ERR_CMD_FAIL               (-4)  //命令执行失败  

PWM_Init

描述：

初始化配置PWM通道，在启动PWM之前，该函数必须调用。

原型：

  1. int WINAPI PWM_Init(int DevHandle, PPWM_CONFIG pConfig);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

pConfig  PWM初始化参数结构体指针。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2pwm.h"  
  6.    
  7.    
  8. int ret;  
  9. //配置PWM相关参数并初始化PWM  
  10. PWM_CONFIG PWMConfig;  
  11. PWMConfig.ChannelMask = 0xFF;//初始化所有通道  
  12. for(int i=0;i<8;i++){  
  13.     PWMConfig.Polarity[i] = 1;//将所有PWM通道都设置为正极性  
  14. }  
  15. for(int i=0;i<8;i++){  
  16.     PWMConfig.Precision[i] = 100;//将所有通道的占空比调节精度都设置为1%  
  17. }  
  18. for(int i=0;i<8;i++){  
  19.     PWMConfig.Phase[i] = 0;//将所有通道的相位都设置为0
  20. }  
  21. for(int i=0;i<8;i++){  
  22.     PWMConfig.Prescaler[i] = 10;//将所有通道的预分频器都设置为10，则PWM输出频率为200MHz/(PWMConfig.Precision*PWMConfig.Prescaler)  
  23. }  
  24. for(int i=0;i<8;i++){  
  25.     PWMConfig.Pulse[i] = PWMConfig.Precision[i]*30/100;//将所有通道的占空比都设置为30%  
  26. }  
  27. //初始化PWM  
  28. ret = PWM_Init(DevHandle,&PWMConfig);  
  29. if(ret != PWM_SUCCESS){  
  30.     printf("Initialize pwm faild!\n");  
  31.     return ret;  
  32. }else{  
  33.     printf("Initialize pwm sunccess!\n");  
  34. }  

PWM_Start

描述：

启动已经初始化的PWM通道，使之输出PWM方波。

原型：

  1. int WINAPI PWM_Start(int DevHandle,unsigned char ChannelMask,unsigned int RunTimeUs);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

ChannelMask  需要输出波形的通道号，每个bit位代表一个通道，bit0对应PWM_CH0，对应位为1，该通道输出波形。

RunTimeUs  输出波形的时间，单位为微妙，启动波形输出之后，RunTimeOfUs微妙之后会停止波形输出，该参数为0，波形会一直输出，直到手动停止，利用该参数可以控制脉冲输出个数。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2pwm.h"  
  6.    
  7.    
  8. int ret;  
  9. //启动PWM,RunTimeUs之后自动停止，利用该特性可以控制输出脉冲个数，脉冲个数=RunTimeUs*200/(PWMConfig.Precision*PWMConfig.Prescaler)  
  10. unsigned int RunTimeUs = 10000;  
  11. ret = PWM_Start(DevHandle,PWMConfig.ChannelMask,RunTimeUs);  
  12. if(ret != PWM_SUCCESS){  
  13.     printf("Start pwm faild!\n");  
  14.     return ret;  
  15. }else{  
  16.     printf("Start pwm sunccess!\n");  
  17. }  

PWM_SetPulse

描述：

设置PWM波占空比，该函数可以在PWM启动之后调用，以便动态改变PWM波形输出占空比。

原型：

  1. int WINAPI PWM_SetPulse(int DevHandle,unsigned char ChannelMask,unsigned short *pPulse);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

ChannelMask  需要输出波形的通道号，每个bit位代表一个通道，bit0对应PWM_CH0，对应位为1，该通道输出波形。

pPulse  波形占空比调节参数首地址。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2pwm.h"  
  6.    
  7.    
  8. int ret;  
  9. //启动PWM,RunTimeUs之后自动停止，利用该特性可以控制输出脉冲个数，脉冲个数=RunTimeUs*200/(PWMConfig.Precision*PWMConfig.Prescaler)  
  10. unsigned int RunTimeUs = 10000;  
  11. ret = PWM_Start(DevHandle,PWMConfig.ChannelMask,RunTimeUs);  
  12. if(ret != PWM_SUCCESS){  
  13.     printf("Start pwm faild!\n");  
  14.     return ret;  
  15. }else{  
  16.     printf("Start pwm sunccess!\n");  
  17. }  
  18. //改变PWM波形占空比  
  19. uint16_t Plse[8];  
  20. for(int i=0;i<100;i+=5){  
  21.     for(int j=0;j<8;j++){  
  22.         Plse[j] = i;  
  23.     }  
  24.     ret = PWM_SetPulse(DevHandle,PWMConfig.ChannelMask,Plse);  
  25.     if(ret != PWM_SUCCESS){  
  26.         printf("Start pwm faild!\n");  
  27.         return ret;  
  28.     }  
  29. }  

PWM_SetPhase

描述：

设置PWM相位，也就是设置PWM计数器的初始值。在PWM频率比较大的时候，设置的参数可能跟实际输出的值误差较大，建议根据实际输出值对该参数进行调整。

原型：

  1. int WINAPI PWM_SetPhase(int DevHandle,unsigned char ChannelMask,unsigned short *pPhase);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

ChannelMask  需要输出波形的通道号，每个bit位代表一个通道，bit0对应PWM_CH0，若对应位为1，则设置该通道的相位参数，否则不设置。

pPhase  波形相位参数首地址。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2pwm.h"  
  6.    
  7.    
  8. int ret;  
  9. //启动PWM,RunTimeUs之后自动停止，利用该特性可以控制输出脉冲个数，脉冲个数=RunTimeUs*200/(PWMConfig.Precision*PWMConfig.Prescaler)  
  10. unsigned int RunTimeUs = 10000;  
  11. ret = PWM_Start(DevHandle,PWMConfig.ChannelMask,RunTimeUs);  
  12. if(ret != PWM_SUCCESS){  
  13.     printf("Start pwm faild!\n");  
  14.     return ret;  
  15. }else{  
  16.     printf("Start pwm sunccess!\n");  
  17. }  
  18. //改变PWM波形相位差 
  19. uint16_t Phase[8];  
  20. for(int j=0;j<8;j++){  
  21.   Phase[j] = j*10;  
  22. }  
  23. ret = PWM_SetPhase(DevHandle,PWMConfig.ChannelMask,Phase);  
  24. if(ret != PWM_SUCCESS){  
  25.     printf("Set pwm phase faild!\n");  
  26.     return ret;  
  27. }  

PWM_SetFrequency

描述：

设置PWM频率，该函数可以在PWM启动之后调用，以便动态改变PWM波形输出频率。频率改变后，占空比保持不变（若改变了占空比精度，可能占空比会有很小的变动），PWM输出频率=200MHz/(Prescaler*Precision)。

原型：

  1. int WINAPI PWM_SetFrequency(int DevHandle,unsigned char ChannelMask,unsigned short *pPrescaler,unsigned short *pPrecision);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice函数获得。

ChannelMask  需要输出波形的通道号，每个bit位代表一个通道，bit0对应PWM_CH0，对应位为1，该通道输出波形。

pPrescaler  PWM时钟预分频器参数指针。

pPrecision  PWM占空比精度调节参数指针。

返回值：

函数执行状态，0为成功，其他值为失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include "usb_device.h"  
  5. #include "usb2pwm.h"  
  6.    
  7.    
  8. int ret;  
  9. //启动PWM,RunTimeUs之后自动停止，利用该特性可以控制输出脉冲个数，脉冲个数=RunTimeUs*200/(PWMConfig.Precision*PWMConfig.Prescaler)  
  10. unsigned int RunTimeUs = 10000;  
  11. ret = PWM_Start(DevHandle,PWMConfig.ChannelMask,RunTimeUs);  
  12. if(ret != PWM_SUCCESS){  
  13.     printf("Start pwm faild!\n");  
  14.     return ret;  
  15. }else{  
  16.     printf("Start pwm sunccess!\n");  
  17. }  
  18. //改变PWM波形输出频率  
  19. uint16_t Precision[8];  
  20. uint16_t  Prescaler[8];  
  21. for(int i=0;i<8;i++){  
  22.     Prescaler[i] = 2;  
  23.     Precision[i] = 100;  
  24. }  
  25. ret = PWM_SetFrequency(DevHandle,PWMConfig.ChannelMask,Prescaler,Precision);  
  26. if(ret != PWM_SUCCESS){  
  27.     printf("Set frequency faild!\n");  
  28.     return ret;  
  29. }
