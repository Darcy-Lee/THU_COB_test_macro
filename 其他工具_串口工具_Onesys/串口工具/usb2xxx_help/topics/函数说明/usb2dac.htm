概述 

本文主要介绍关于USB2XXX的ADAC操作相关API，在调用这些函数前需要调用设备操作相关函数，比如扫描设备，打开设备等，详情请查看《设备操作相关函数说明 》。

下面是本文函数可能会用到的一些宏定义：

  1. //定义函数返回错误代码  
  2. #define DAC_SUCCESS                          (0)   //函数执行成功  
  3. #define DAC_ERR_NOT_SUPPORT       (-1)  //适配器不支持该函数  
  4. #define DAC_ERR_USB_WRITE_FAIL  (-2)  //USB写数据失败  
  5. #define DAC_ERR_USB_READ_FAIL    (-3)  //USB读数据失败  
  6. #define DAC_ERR_CMD_FAIL               (-4)  //命令执行失败  
  7. #define DAC_ERR_CH_NO_INIT          (-5)  //该通道未初始化  

DAC_Init

描述：

初始化配置DAC并根据配置数据输出DAC波形。

原型：

  1. int WINAPI DAC_Init(int DevHandle,unsigned short CycleUs,unsigned short *pWaveData,unsigned short WaveDataNum);  

参数：

DevHandle  设备句柄，本质为设备序号的低4字节，可以通过调用USB_ScanDevice 函数或得。

CycleUs  DAC数据输出间隔时间，单位为0.25微妙，比如该参数传入4，那么DAC数据输出的间隔时间就是0.25*4=1us，波形输出的周期= 0.25*CycleUs*WaveDataNum微秒。

pWaveData  DAC输出数据指针，DAC为12bit转换，所以单个数据最大值为4095，对应3.3V，最多只能输出512个数据，最少输出2个数据。

WaveDataNum DAC输出数据个数，最小为2，最大为512.

返回值：

初始化DAC的状态，0-配置成功，其他值，配置失败。

示例：

  1. #include <stdlib.h>  
  2. #include <stdio.h>  
  3. #include <string.h>  
  4. #include <stdint.h>  
  5. #include "usb_device.h"  
  6. #include "usb2dac.h"  
  7. #include <math.h>  
  8. int main(int argc, const char* argv[])  
  9. {  
  10.     DEVICE_INFO DevInfo;  
  11.     int DevHandle[10];  
  12.     bool state;  
  13.     int ret;  
  14.     unsigned short DACBuffer[512];//缓冲区数据不能大于512  
  15.     //扫描查找设备  
  16.     ret = USB_ScanDevice(DevHandle);  
  17.     if(ret <= 0){  
  18.         printf("No device connected!\n");  
  19.         return 0;  
  20.     }  
  21.     //打开设备  
  22.     state = USB_OpenDevice(DevHandle[0]);  
  23.     if(!state){  
  24.         printf("Open device error!\n");  
  25.         return 0;  
  26.     }  
  27.     //获取固件信息  
  28.     char FuncStr[256]={0};  
  29.     state = DEV_GetDeviceInfo(DevHandle[0],&DevInfo,FuncStr);  
  30.     if(!state){  
  31.         printf("Get device infomation error!\n");  
  32.         return 0;  
  33.     }else{  
  34.         printf("Firmware Info:\n");  
  35.         printf("    Name:%s\n",DevInfo.FirmwareName);  
  36.         printf("    Build Date:%s\n",DevInfo.BuildDate);  
  37.         printf("    Firmware Version:v%d.%d.%d\n",(DevInfo.FirmwareVersion>>24)&0xFF,(DevInfo.FirmwareVersion>>16)&0xFF,DevInfo.FirmwareVersion&0xFFFF);  
  38.         printf("    Hardware Version:v%d.%d.%d\n",(DevInfo.HardwareVersion>>24)&0xFF,(DevInfo.HardwareVersion>>16)&0xFF,DevInfo.HardwareVersion&0xFFFF);  
  39.         printf("    Functions:%08X\n",DevInfo.Functions);  
  40.         printf("    Functions:%s\n",FuncStr);  
  41.     }  
  42.     //产生方波信号  
  43.     for(int i=0;i<256;i++){  
  44.         DACBuffer[i] = 0xFFF;  
  45.     }  
  46.     for(int i=256;i<512;i++){  
  47.         DACBuffer[i] = 0x0;  
  48.     }  
  49.     //初始化ADC,初始化配置后会自动输出波形  
  50.     int TimeOfUs = 4;//每个数据输出的时间周期 = TimeOfUs*0.25 us  
  51.     ret = DAC_Init(DevHandle[0],TimeOfUs,DACBuffer,512);//周期 = 0.25*TimeOfUs*512 us  
  52.     if(ret != DAC_SUCCESS){  
  53.         printf("Init dac error!\n");  
  54.         return 0;  
  55.     }  
  56.     getchar();  
  57.     //输出正弦信号  
  58.     float dcVoltage = 1;    //直流分量  
  59.     float amplitude = 2.3;  //幅值，幅值加上直流分量不能大于3.3V  
  60.     int DataNum=64;//周期 = 0.25*TimeOfUs*DataNum us  
  61.     for(int i=0;i<DataNum;i++){  
  62.         DACBuffer[i] = ((dcVoltage+amplitude*(1+sin(2*3.141592653589793*i/DataNum))/2.0)*4095)/3.3;  
  63.     }  
  64.     //初始化ADC,初始化配置后会自动输出波形  
  65.     ret = DAC_Init(DevHandle[0],TimeOfUs,DACBuffer,DataNum);//周期 = 0.25*TimeOfUs*512 us  
  66.     if(ret != DAC_SUCCESS){  
  67.         printf("Init dac error!\n");  
  68.         return 0;  
  69.     }  
  70.     getchar();  
  71.     //输出三角波信号  
  72.     for(int i=0;i<DataNum/2;i++){  
  73.         DACBuffer[i] = (dcVoltage+amplitude*(i/(DataNum/2.0)))*4095.0/3.3;  
  74.     }  
  75.     for(int i=0;i<DataNum/2;i++){  
  76.         DACBuffer[i+DataNum/2] = (dcVoltage+amplitude*((DataNum/2.0-i)/(DataNum/2.0)))*4095.0/3.3;  
  77.     }  
  78.     getchar();  
  79.     //停止数据输出  
  80.     DAC_Stop(DevHandle[0]);  
  81.     //关闭设备  
  82.     USB_CloseDevice(DevHandle[0]);  
  83.     return 0;  
  84. }
